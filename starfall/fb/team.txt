fbteams = {}


function newteam(name)
    fbteams[name] = {}
end

function addplayer(name, player)
    table.insert(fbteams[name], player)
end

function removeplayer(name, player)
    for k, v in pairs(fbteams[name]) do
        if v == player then
            table.remove(fbteams[name], k)
            return true
        end
    end
    return false
end    

hook.add("Command", "Football_teams", function(cmd, args, text)
    if cmd ~= "team" then return end
    local action = table.remove(args, 1)
    
    if action == "create" then
        local name = args[1]
        if fbteams[name] then
            print("\"" .. name .. "\" already exists!")
        else
            newteam(name)
            print("Created team \"" .. name .. "\"")
        end
    elseif action == "add" then
        local name = table.remove(args, 1)
        local players = args
        for k, v in pairs(players) do
            local ply = pw.findPlayerByName(v)
            if ply then
                print("Added \"" .. ply:getName() .. "\" to \"" .. name .. "\"")
                addplayer(name, pw.findPlayerByName(v))
            else
                print("Can't find \"" .. v .. "\"")
            end
        end
    elseif action == "remove" then
        local name = table.remove(args, 1)
        local players = args
        for k, v in pairs(players) do
            local ply = pw.findPlayerByName(v)
            if ply then
                if removeplayer(name, ply) then
                    print("Removed \"" .. ply:getName() .. "\" from \"" .. name .. "\"")
                end
            else
                print("Can't find \"" .. v .. "\"")
            end
        end
    elseif action == "debug" then
        printTable(fbteams)
    end
end)
